# LangChain Implementation - Key Components

class SimpleResearchAssistant:
    def __init__(self):
        self.llm = ChatAnthropic(model="claude-3-5-sonnet-20241022", temperature=0)
    
    def create_analysts(self, topic: str, num_analysts: int = 3):
        """Create diverse analyst personas using structured output"""
        structured_llm = self.llm.with_structured_output(Perspectives)
        
        system_message = f"""Create {num_analysts} different AI analyst personas 
        for research topic: {topic}. Each should have unique expertise."""
        
        result = structured_llm.invoke([
            SystemMessage(content=system_message),
            HumanMessage(content="Generate the analysts.")
        ])
        return result.analysts
    
    def conduct_interview(self, analyst: Analyst, topic: str, max_turns: int = 3):
        """Sequential interview with Wikipedia search"""
        conversation_history = []
        
        for turn in range(1, max_turns + 1):
            # Generate question
            question = question_generation_chain.invoke({
                "analyst_persona": analyst.persona,
                "topic": topic,
                "turn_number": turn,
                "conversation_history": "\n".join(conversation_history)
            })
            
            # Search Wikipedia
            research_context = search_wikipedia(f"{topic} {question.content}")
            
            # Generate expert response
            answer = expert_response_chain.invoke({
                "topic": topic,
                "analyst_persona": analyst.persona,
                "question": question.content,
                "research_context": research_context
            })
            
            conversation_history.extend([
                f"Analyst: {question.content}",
                f"Expert: {answer.content}"
            ])
        
        return conversation_history
# LangGraph Implementation - Key Components

from langgraph.graph import START, END, StateGraph
from langgraph.constants import Send
from typing_extensions import TypedDict

class ResearchGraphState(TypedDict):
    topic: str
    analysts: List[Analyst]
    sections: Annotated[list, operator.add]  # Auto-merge with +
    final_report: str

def initiate_all_interviews(state: ResearchGraphState):
    """Start parallel interviews using Send API"""
    return [Send("conduct_interview", {
        "analyst": analyst,
        "messages": [HumanMessage(content=f"Research topic: {state['topic']}")]
    }) for analyst in state["analysts"]]

def build_research_graph():
    """Build the main research graph"""
    main_builder = StateGraph(ResearchGraphState)
    
    # Add nodes
    main_builder.add_node("create_analysts", create_analysts)
    main_builder.add_node("human_feedback", human_feedback)
    main_builder.add_node("conduct_interview", interview_graph)  # Sub-graph
    main_builder.add_node("write_report", write_report)
    main_builder.add_node("finalize_report", finalize_report)
    
    # Define workflow
    main_builder.add_edge(START, "create_analysts")
    main_builder.add_edge("create_analysts", "human_feedback")
    main_builder.add_conditional_edges("human_feedback", initiate_all_interviews, 
                                     ["create_analysts", "conduct_interview"])
    main_builder.add_edge("conduct_interview", "write_report")
    main_builder.add_edge("write_report", "finalize_report")
    main_builder.add_edge("finalize_report", END)
    
    return main_builder.compile(
        interrupt_before=['human_feedback'],  # Human-in-the-loop
        checkpointer=MemorySaver()           # State persistence
    )
